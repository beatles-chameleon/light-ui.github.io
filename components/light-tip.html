light-tip

tip

通用提示

属性

属性名 类型 必填 默认值 说明
show Boolean 否 false 提示框是否显示
direction String 否 top 提示框所在位置，可选值top/right/bottom/left
offset-left Number 否 小三角之Tip左边距离
offset-top Number 否 小三角之Tip顶部距离
offset-right Number 否 小三角之Tip右边距离
offset-bottom Number 否 小三角之Tip底部距离
c-bind:close EventHandle 点击关闭按钮触发

注意：Tip小三角默认是居中显示的，当direction值为top或bottom时，如果offset-left或offset-right有值且不是NaN，则小三角的距离为传入值

示例

<template>
  <page title="light-tip">
    <c-header title="light-tip"></c-header>
    <view class="tip-eg">
      <view style="{{tipStyle}}">
        <light-tip 
          show="{{show}}" 
          direction="{{direction}}" 
          c-bind:close="closeTip" 
        >
          <text style="color: #fff">Awesome!</text>
        </light-tip>
      </view>
      <text class="tip-btn" c-bind:tap="showTip">light-ui tip</text>
      <view class="btn-wrap">
        <light-button class="btn"  c-bind:onClick="showTip2Top" text="top"></light-button>
        <light-button class="btn"  c-bind:onClick="showTip2Bottom" text="bottom"></light-button>
        <light-button class="btn"  c-bind:onClick="showTip2Left" text="left"></light-button>
        <light-button class="btn"  c-bind:onClick="showTip2Right" text="right"></light-button>
      </view>
    </view>

  </page>
</template>

<script>
import cml from 'chameleon-api';

class LightBartips   {

  data = {
    show: false,
    duration: 2000,
    direction: 'top'
  }
  computed = {
    tipStyle() {
      let style;
      switch (this.direction) {
        case "top":
          style =
            "position:absolute;top:0;left: 260cpx;";
          break;
        case "bottom":
          style =
            "position:absolute;top:170cpx;left: 260cpx;";
          break;
        case "left":
          style = "position:absolute;left: 0cpx;top:93cpx;";
          break;
        case "right":
          style = "position:absolute;left:480cpx;top:93cpx;";
          break;
      }
      return style;
    }
  }

  methods = {
    showTip() {
      this.show = true
    },
    closeTip() {
      this.show = false;
    },
     showTip2Top(){
      this.direction = "top";
      this.show = true;
    },
    showTip2Bottom(){
      this.direction ="bottom";
      this.show = true;
    },
    showTip2Left(){
      this.direction = "left";
      this.show = true;
    },
    showTip2Right(){
      this.direction = "right";
      this.show = true;
    }

  }

}

export default new LightBartips();
</script>
